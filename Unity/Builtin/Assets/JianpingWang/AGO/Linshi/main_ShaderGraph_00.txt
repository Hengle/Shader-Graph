Shader "Main/Diffuse Texture"
{
   Properties
{
     _MainTex ("Base (RGB) , 2D) = "white" {}
}
   subshader
{
     Tags { "RenderType" = "Opaque" }
     LOD 200
     
     CGPROGRAM
     #pragram surface surf Lambert

     sampler2D _MainTex;
     
     struct Input 
     {
          float2 uv_MainTex;
     };
     
     void surf (Input IN, inout SurfaceOutput o)
     {
          half4 = tex2D (_MainTex, In.uv_MainTex);
          o.Albedo = c.rgb;
          o.Alpha = c.a;
      }
     ENDCG
}
   FallBack "Diffuse"
}
